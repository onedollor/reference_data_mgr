20 def detect_format(self, file_path: str, sample_size: int = 8192) -> Dict[str, Any]: 

21 """ 

22 Detect CSV format parameters from a file 

23  

24 Args: 

25 file_path: Path to the CSV file 

26 sample_size: Number of bytes to read for detection 

27  

28 Returns: 

29 Dictionary containing detected format parameters 

30 """ 

31 try: 

32 # Read file with encoding detection 

33 with open(file_path, 'rb') as file: 

34 raw_data = file.read(sample_size) 

35 encoding_result = chardet.detect(raw_data) 

36 encoding = encoding_result['encoding'] if encoding_result['confidence'] > 0.7 else 'utf-8' 

37 

38 # Read file content with detected encoding 

39 with open(file_path, 'r', encoding=encoding, errors='ignore') as file: 

40 content = file.read(sample_size) 

41 

42 if not content.strip(): 

43 raise ValueError("File appears to be empty") 

44 

45 lines = content.split('\n') 

46 if len(lines) < 2: 

47 raise ValueError("File must have at least 2 lines for detection") 

48 

49 # Detect format parameters 

50 column_delimiter = self._detect_column_delimiter(content, lines) 

51 row_delimiter = self._detect_row_delimiter(content) 

52 text_qualifier = self._detect_text_qualifier(content, column_delimiter) 

53 has_header = self._detect_header(lines, column_delimiter, text_qualifier) 

54 

55 # For trailer detection we must inspect the real last non-empty lines of the FULL file, not just the sample window. 

56 trailer_line = None 

57 has_trailer = False 

58 try: 

59 with open(file_path, 'r', encoding=encoding, errors='ignore') as f_full: 

60 full_lines = [ln.rstrip('\n\r') for ln in f_full] 

61 # Collect last 200 non-empty lines to minimize memory while covering edge cases (footers sometimes multi-line) 

62 non_empty_full = [l for l in full_lines if l.strip()] 

63 if len(non_empty_full) >= 2: 

64 last_line_full = non_empty_full[-1] 

65 prev_line_full = non_empty_full[-2] 

66 # Derive typical column count from first 100 non-empty data lines (excluding potential header and last line) 

67 sample_for_mode = [] 

68 start_index = 1 if has_header else 0 

69 for l in non_empty_full[start_index: min(start_index+100, len(non_empty_full)-1)]: 

70 try: 

71 sample_for_mode.append(len(self._parse_row(l, column_delimiter, text_qualifier))) 

72 except Exception: 

73 pass 

74 typical_cols = None 

75 if sample_for_mode: 

76 try: 

77 from statistics import mode 

78 typical_cols = mode(sample_for_mode) 

79 except Exception: 

80 typical_cols = Counter(sample_for_mode).most_common(1)[0][0] 

81 try: 

82 last_cols = len(self._parse_row(last_line_full, column_delimiter, text_qualifier)) 

83 prev_cols = len(self._parse_row(prev_line_full, column_delimiter, text_qualifier)) 

84 except Exception: 

85 last_cols = prev_cols = None 

86 # Trailer per business rule: last line column count differs from previous AND previous matches typical columns 

87 if last_cols is not None and prev_cols is not None and last_cols != prev_cols: 

88 if typical_cols is None or prev_cols == typical_cols: 

89 has_trailer = True 

90 trailer_line = last_line_full 

91 except Exception: 

92 # Fail silent; fallback to sample-based basic rule 

93 if not has_trailer: 

94 has_trailer, trailer_line = self._detect_trailer(lines, column_delimiter, text_qualifier) 

95 

96 # Parse first few rows to get column info 

97 sample_data = self._parse_sample_data(lines[:10], column_delimiter, text_qualifier) 

98 

99 # Just store the actual trailer line text if detected 

100 

101 result = { 

102 'encoding': encoding, 

103 'encoding_confidence': encoding_result['confidence'], 

104 'column_delimiter': column_delimiter, 

105 'row_delimiter': row_delimiter, 

106 'text_qualifier': text_qualifier, 

107 'has_header': has_header, 

108 'has_trailer': has_trailer, 

109 'estimated_columns': len(sample_data[0]) if sample_data else 0, 

110 'sample_rows': len(lines), 

111 'detection_confidence': self._calculate_confidence(content, column_delimiter, text_qualifier), 

112 'sample_data': sample_data[:3], # First 3 rows as sample 

113 'header_delimiter': column_delimiter, # Same as column delimiter typically 

114 'skip_lines': 0, # Can be enhanced to detect skip lines 

115 'trailer_line': trailer_line if has_trailer else None 

116 } 

117 

118 return result 

119 

120 except Exception as e: 

121 return { 

122 'error': str(e), 

123 'column_delimiter': ',', 

124 'row_delimiter': '\n', 

125 'text_qualifier': '"', 

126 'has_header': True, 

127 'has_trailer': False, 

128 'header_delimiter': ',', 

129 'skip_lines': 0, 

130 'detection_confidence': 0.0 

131 }

133 def _detect_column_delimiter(self, content: str, lines: List[str]) -> str: 

134 """Detect the column delimiter""" 

135 delimiter_scores = {} 

136 

137 for delimiter in self.common_delimiters: 

138 score = 0 

139 delimiter_counts = [] 

140 

141 # Check consistency across lines 

142 for line in lines[:10]: # Check first 10 lines 

143 if line.strip(): 

144 count = line.count(delimiter) 

145 delimiter_counts.append(count) 

146 

147 if delimiter_counts: 

148 # Score based on consistency and frequency 

149 avg_count = sum(delimiter_counts) / len(delimiter_counts) 

150 consistency = 1.0 - (max(delimiter_counts) - min(delimiter_counts)) / (max(delimiter_counts) + 1) 

151 score = avg_count * consistency 

152 delimiter_scores[delimiter] = score 

153 

154 # Return delimiter with highest score 

155 if delimiter_scores: 

156 best_delimiter = max(delimiter_scores.keys(), key=lambda x: delimiter_scores[x]) 

157 return best_delimiter 

158 

159 return ',' # Default fallback

161 def _detect_row_delimiter(self, content: str) -> str: 

162 """Detect the row delimiter""" 

163 crlf_count = content.count('\r\n') 

164 lf_count = content.count('\n') - crlf_count 

165 cr_count = content.count('\r') - crlf_count 

166 

167 counts = {'\r\n': crlf_count, '\n': lf_count, '\r': cr_count} 

168 

169 # Return the most common row delimiter 

170 best_delimiter = max(counts.keys(), key=lambda x: counts[x]) 

171 return best_delimiter if counts[best_delimiter] > 0 else '\n'

173 def _detect_text_qualifier(self, content: str, column_delimiter: str) -> str: 

174 """Detect the text qualifier""" 

175 # Look for quoted fields 

176 patterns = [ 

177 rf'{re.escape(column_delimiter)}"[^"]*"', # ,"text" 

178 rf'{re.escape(column_delimiter)}\'[^\']*\'', # ,'text' 

179 rf'^"[^"]*"', # "text" at start 

180 rf'^\'[^\']*\'', # 'text' at start 

181 ] 

182 

183 quote_scores = {'"': 0, "'": 0, '': 0} 

184 

185 for pattern in patterns: 

186 if '"' in pattern: 

187 quote_scores['"'] += len(re.findall(pattern, content, re.MULTILINE)) 

188 elif "'" in pattern: 

189 quote_scores["'"] += len(re.findall(pattern, content, re.MULTILINE)) 

190 

191 # Check for fields that contain the delimiter (would need quotes) 

192 delimiter_in_field_pattern = rf'[^{re.escape(column_delimiter)}]*{re.escape(column_delimiter)}[^{re.escape(column_delimiter)}]*' 

193 potential_quoted_fields = len(re.findall(delimiter_in_field_pattern, content)) 

194 

195 if quote_scores['"'] > quote_scores["'"]: 

196 return '"' 

197 elif quote_scores["'"] > 0: 

198 return "'" 

199 elif potential_quoted_fields > 0: 

200 return '"' # Default to double quotes if fields likely contain delimiters 

201 

202 return '' # No text qualifier needed

204 def _detect_header(self, lines: List[str], column_delimiter: str, text_qualifier: str) -> bool: 

205 """Detect if the file has a header row""" 

206 if len(lines) < 2: 

207 return True # Assume header if only one line 

208 

209 try: 

210 # Parse first two rows 

211 first_row = self._parse_row(lines[0], column_delimiter, text_qualifier) 

212 second_row = self._parse_row(lines[1], column_delimiter, text_qualifier) 

213 

214 if len(first_row) != len(second_row): 

215 return True # Different column counts suggest header 

216 

217 # Check data types - headers are usually strings, data has mixed types 

218 first_row_types = self._analyze_data_types(first_row) 

219 second_row_types = self._analyze_data_types(second_row) 

220 

221 # If first row is all strings and second has numbers, likely header 

222 if (first_row_types.count('string') == len(first_row) and 

223 second_row_types.count('number') > 0): 

224 return True 

225 

226 # Check for common header patterns 

227 header_indicators = ['id', 'name', 'date', 'time', 'code', 'description', 'value', 'amount'] 

228 first_row_lower = [str(cell).lower().strip() for cell in first_row] 

229 

230 header_matches = sum(1 for cell in first_row_lower if any(indicator in cell for indicator in header_indicators)) 

231 if header_matches > len(first_row) * 0.3: # 30% of columns match header patterns 

232 return True 

233 

234 return False # Probably no header 

235 

236 except: 

237 return True # Default to assuming header on parse errors

239 def _detect_trailer(self, lines: List[str], column_delimiter: str, text_qualifier: str) -> tuple[bool, Optional[str]]: 

240 """Simple trailer detection per spec: examine last non-empty line. 

241 If its column count differs from the line immediately above (which matches typical data), 

242 classify it as trailer. 

243 Returns (has_trailer, trailer_line_text). 

244 """ 

245 non_empty = [l for l in lines if l.strip()] 

246 if len(non_empty) < 2: 

247 return False, None 

248 last_line = non_empty[-1] 

249 prev_line = non_empty[-2] 

250 try: 

251 last_cols = len(self._parse_row(last_line, column_delimiter, text_qualifier)) 

252 prev_cols = len(self._parse_row(prev_line, column_delimiter, text_qualifier)) 

253 except Exception: 

254 return False, None 

255 if prev_cols > 0 and last_cols != prev_cols: 

256 return True, last_line 

257 return False, None

261 """Parse a single row with given format parameters using proper CSV parsing""" 

262 import io

264 reader = csv.reader( 

265 io.StringIO(line), 

266 delimiter=delimiter, 

267 quotechar=text_qualifier if text_qualifier else None

269 try:

271 except StopIteration:

274 def _parse_sample_data(self, lines: List[str], delimiter: str, text_qualifier: str) -> List[List[str]]: 

275 """Parse sample data rows"""

286 def _analyze_data_types(self, row: List[str]) -> List[str]: 

287 """Analyze data types in a row""" 

288 types = [] 

289 for cell in row: 

290 cell_str = str(cell).strip() 

291 if not cell_str: 

292 types.append('empty') 

293 elif cell_str.replace('.', '').replace('-', '').replace('+', '').isdigit(): 

294 types.append('number') 

295 elif re.match(r'^\d{4}-\d{2}-\d{2}', cell_str): 

296 types.append('date') 

297 else: 

298 types.append('string') 

299 return types

301 def _calculate_confidence(self, content: str, delimiter: str, text_qualifier: str) -> float: 

302 """Calculate confidence score for the detection""" 

303 score = 0.5 # Base score 

304 

305 # Check delimiter consistency 

306 lines = content.split('\n')[:10] 

307 delimiter_counts = [line.count(delimiter) for line in lines if line.strip()] 

308 if delimiter_counts: 

309 consistency = 1.0 - (max(delimiter_counts) - min(delimiter_counts)) / (max(delimiter_counts) + 1) 

310 score += consistency * 0.3 

311 

312 # Check if quotes are balanced 

313 if text_qualifier: 

314 quote_count = content.count(text_qualifier) 

315 if quote_count % 2 == 0: # Even number suggests balanced quotes 

316 score += 0.2 

317 

318 return min(score, 1.0)
